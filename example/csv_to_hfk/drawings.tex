% Iterate over string
% Taken from https://tex.stackexchange.com/q/359189/216213
\def\xloop<#1#2>{%
  \ifx\relax#1
  \else
    \ifthenelse{\equal{#1}{0}}
      {\mathord{\hspace{-.5pt}\cdot\hspace{-.5pt}}}
      {\mathord{\hspace{-.4pt}\raisebox{.4pt}{$\shortmid$}\hspace{-.4pt}}}%{\hspace{.2pt}\mathord\mhyphen\hspace{.2pt}}%
    \xloop<#2>%
  \fi}
\newcommand{\nodeLabel}[1]{$\xloop<#1\relax>$}
